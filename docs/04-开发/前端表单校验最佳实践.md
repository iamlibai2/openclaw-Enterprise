```typescript
interface FieldRule {
  required?: boolean
  minLength?: number
  maxLength?: number
  pattern?: RegExp
  messages: {
    required?: string
    minLength?: string
    maxLength?: string
    pattern?: string
  }
}
```

## 使用方式

### 基础用法

```typescript
import { createFormRules, sanitizeData } from '../utils/rules'

// 创建 Element Plus 表单规则
const formRules = createFormRules({
  name: 'modelName',
  provider: 'providerName'
})
```

### 动态校验规则

```typescript
import { type FormRules } from 'element-plus'

// 基础规则
const baseRules = createFormRules({
  model_name: 'modelName'
})

// 动态规则（带唯一性校验）
const formRules = computed<FormRules>(() => ({
  ...baseRules,
  model_name: [
    ...baseRules.model_name,
    {
      validator: (rule, value, callback) => {
        // 编辑模式：原值不检查
        if (isEdit.value && currentModel.value?.model_name === value) {
          callback()
          return
        }
        // 检查唯一性
        const exists = models.value.some(m => m.model_name === value)
        if (exists) {
          callback(new Error('模型名称已存在'))
        } else {
          callback()
        }
      },
      trigger: 'blur'
    }
  ]
}))
```

## 数据清理

### sanitizeData 函数

```typescript
export function sanitizeData(data: Record<string, any>): Record<string, any> {
  const result: Record<string, any> = {}
  for (const [key, value] of Object.entries(data)) {
    if (typeof value === 'string') {
      result[key] = value.trim()  // 去除首尾空格
    } else if (value !== null && typeof value === 'object' && !Array.isArray(value)) {
      result[key] = sanitizeData(value)  // 递归处理嵌套对象
    } else {
      result[key] = value
    }
  }
  return result
}
```

### 使用示例

```typescript
async function submitForm() {
  // 清理数据
  const cleanData = sanitizeData({
    name: formData.name,
    model_name: formData.model_name,
    api_key: formData.api_key
  })

  // 提交到后端
  await api.create(cleanData)
}
```

## 浏览器自动填充问题

### 问题

当表单包含 `type="password"` 输入框时，浏览器会自动填充登录凭据。

### 解决方案

```vue
<!-- 普通输入框 -->
<el-input v-model="formData.name" autocomplete="off" />

<!-- 密码输入框 -->
<el-input
  v-model="formData.api_key"
  type="password"
  autocomplete="new-password"
/>
```

### 属性说明

| 属性 | 值 | 说明 |
|------|-----|------|
| `autocomplete` | `off` | 禁用自动填充（普通输入框） |
| `autocomplete` | `new-password` | 新密码字段，不自动填充 |
