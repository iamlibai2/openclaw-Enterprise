  if (evt.event !== 'chat' || !selectedSessionKey.value) return

  if (payload.state === 'delta') {

def _get_gateway_token() -> str:
    # 1. 环境变量
    env_token = os.environ.get('GATEWAY_AUTH_TOKEN')
    if env_token:
        return env_token

    # 2. 从 openclaw.json 获取
    try:
        config_path = Path.home() / '.openclaw' / 'openclaw.json'
        if config_path.exists():
            with open(config_path, 'r') as f:
                data = json.load(f)
                gateway = data.get('gateway', {})
                auth = gateway.get('auth', {})

| `aborted` | 被中止 | 保存已输出内容 |
| `error` | 出错 | 显示错误信息 |

### 2. Markdown 渲染优化

**背景**：Chat 页面消息只显示原始文本，Markdown 表格、代码块等无法正确渲染。

#### 2.2 技术方案

**依赖库**：

- `dompurify` - HTML 安全清理（新增）

**核心实现**：

```typescript
// markdown.ts
export function renderMessageContent(text: string): string {
  // 1. 移除 thinking 标签
  const cleanedText = stripThinkingTags(text)

  // 2. 检测文本方向
  const dir = detectTextDirection(cleanedText)

  // 3. 检测文本表格
  const isTable = isPlainTextTable(cleanedText)

  // 4. Markdown 渲染
  const html = toSanitizedMarkdownHtml(cleanedText)

  // 5. 包装容器
  return `<div class="${isTable ? 'text-table' : 'markdown-body'}" dir="${dir}">${html}</div>`
}
```

#### 2.3 关键问题：Vue scoped 样式

**问题**：`<style scoped>` 中的样式不会应用到 `v-html` 渲染的动态内容。

**解决**：将 Markdown 样式放到非 scoped 的 `<style>` 块中：

```vue
<style scoped>
/* 组件内部样式 */
.message { ... }
</style>

<!-- 非 scoped：应用到 v-html 内容 -->
<style>
.markdown-body table { border-collapse: collapse; }
.code-block-wrapper { background: #1e1e1e; }
</style>
```

## 6. 前端校验架构完整改造

**时间**：18:00-19:00

**背景**：发现模型名称保存后出现 `"\t MiniMax-M2.5"` 格式错误，输入未清理首尾空格和制表符。

### 6.1 问题分析

```
配置文件中出现：
"id": "\t MiniMax-M2.5",
"name": " MiniMax-M2.5",

原因：前端输入未清理，后端也未做防御性处理
```

### 6.2 解决方案

#### 新增 `sanitizeData` 函数

```typescript
// utils/rules.ts

/**
 * 清理输入数据 - 去除字符串首尾空格
 */
export function sanitizeData(data: Record<string, any>): Record<string, any> {
  const result: Record<string, any> = {}
  for (const [key, value] of Object.entries(data)) {
    if (typeof value === 'string') {
      result[key] = value.trim()
    } else if (value !== null && typeof value === 'object' && !Array.isArray(value)) {
      result[key] = sanitizeData(value)  // 递归处理嵌套对象
    } else {
      result[key] = value
    }
  }
  return result
}
```
